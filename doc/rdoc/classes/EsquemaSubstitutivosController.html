def create
  @esquema_substitutivo = EsquemaSubstitutivo.new(esquema_substitutivo_params)

  if @esquema_substitutivo.save
    redirect_to @esquema_substitutivo, notice: 'Esquema substitutivo was successfully created.'
  else
    render :new
  end
end
