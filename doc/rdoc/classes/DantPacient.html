def set_frascos 
  case self.frascos_diarios
    when 1..33
      self.frascos_mensais = 1
    when 34..66
      self.frascos_mensais = 2
    when 67..99
      self.frascos_mensais = 3
    when 100..132
      self.frascos_mensais = 4
  end
end
